S3bucketpolicy.tf


resource "aws_s3_bucket_policy" "allow_access" {
  bucket = "${aws_s3_bucket.mys3.id}
  policy = data.aws_iam_policy_document.allow_access
}

data "aws_iam_policy_document" "allow_access" {
  statement {
    principals {
      type        = "AWS"
      
    }

    actions = [
      "s3:GetObject",
      "s3:ListBucket",
    ]

   }
}
